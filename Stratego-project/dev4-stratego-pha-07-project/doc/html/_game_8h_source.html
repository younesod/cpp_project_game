569inline void Game::applyMove(const Position& start, const Direction &direction, const int &nbSteps)
570{
571 if(this->state!=State::CHOOSE_PIECE){
572 throw invalid_argument("the state is supposed to be CHOOSE_PIECE");
573 }
574 Position nextDir=start.nextPosition(direction,nbSteps);
575 if(this->board.getSquare(nextDir).isFree()){ //if next piece is free
576 this->board.put(this->board.getPiece(start),nextDir);
577 this->board.remove(start);
578 this->state = State::MOVE_PIECE;
579 }else if(this->board.getSquare(start).getPiece()->getColorPiece() != this->board.getSquare(nextDir).getPiece()->getColorPiece()){
580 this->board.getPiece(nextDir)->toggleVisibility(); //la piece adverse est revelé
581 if(this->board.getPiece(start)->isStronger(this->board.getPiece(nextDir))){
582 if(this->board.getPiece(nextDir)->getSymbol()==0){
583 if(this->currentPlayer.getPColor()==RED) this->playerBlue.getDeckP().setFlag(nullptr);
584 else this->playerRed.getDeckP().setFlag(nullptr);
585 }
586 this->board.remove(nextDir);
587 this->board.put(this->board.getPiece(start),nextDir);
588 this->board.remove(start);
589 this->state = State::MOVE_PIECE;
590 }else if(this->board.getPiece(start)->hasSameSymbol(this->board.getPiece(nextDir))){
591 this->board.remove(nextDir);
592 this->board.remove(start);
593 this->state = State::MOVE_PIECE;
594 }else if(!this->board.getPiece(start)->isStronger(this->board.getPiece(nextDir))){
595 this->board.remove(start);
596 this->state = State::MOVE_PIECE;
597 }
598 }

600inline void Game::move(const Position& start, const Direction direction, const int nbSteps=1){
601 if(this->board.getPiece(start) == nullptr || !this->board.isInside(start)){
602 throw invalid_argument("the position selected is empty or not inside the board or the direction ");
603 }
604 Position nextDir=start.nextPosition(direction,nbSteps);
605 if(this->board.getSquare(nextDir).isLand() && this->board.isInside(nextDir)
606 && !this->board.getSquare(nextDir).isMyPiece(this->currentPlayer.getPColor())){
607 if(this->board.getSquare(start).getPiece()->getTypeMove()==Move::NORMAL){
608 if(this->board.isMyPiece(start,this->currentPlayer.getPColor())){
609 this->applyMove(start,direction,nbSteps);
610 this->setTurnEnd(this->currentPlayer.getPColor());
611 notifyAll(state);
612 }
613 }
614 else if(this->board.getSquare(start).getPiece()->getTypeMove()==Move::INFINITE){
615 Position posPossible=canPass(start,direction);
616 if(this->board.isMyPiece(start,this->currentPlayer.getPColor()) && isMovePossible(direction,nextDir,posPossible)){
617 this->applyMove(start,direction,nbSteps);
618 this->setTurnEnd(this->currentPlayer.getPColor());
619 notifyAll(state);
620 }
621 }
622 }
